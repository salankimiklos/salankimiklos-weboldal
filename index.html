# 2_1a)
lmap = lambda x,r : r*x*(1-x)
lmap_der = lambda x,r: r*(1-2*x)
err=1e-9
N=500
itermax=500
r=np.linspace(3,4,N)
h=np.zeros(N)

for i in range(N):
    y=[]
    y.append(np.random.rand())
    for k in range(itermax):
        x=lmap(y[-1],r[i])
        if abs(x-y[-1])<err:
            break
        y.append(x)
        
    y=np.array(y)
    n=len(y)
    kihagyas=5
    for j in range(kihagyas,n):
        h[i]=h[i]+np.log(abs(lmap_der(y[j],r[i])))
    h[i]=h[i]/(n-kihagyas)
    
plt.figure()
plt.plot(r,h,'C2')
plt.xlabel("r",fontsize=14)
plt.ylabel("h",fontsize=14)
plt.title("Liapunov exponents of logistic map",fontsize=15)
plt.grid()
plt.savefig("2_1_Liap_3-4.pdf")
plt.show()

# 2_1d)
# According to point a) for r=4 -> h=0.6891895618920415
r=4
epsz=1e-7
x0=np.random.rand()
y0=x0+epsz
N=25
n=np.arange(N)
x=np.zeros(N)
y=np.zeros(N)
dif=np.zeros(N)
x[0]=x0
y[0]=y0
dif[0]=abs(x0-y0)
for i in range(1,N):
    x[i]=lmap(x[i-1],r)
    y[i]=lmap(y[i-1],r)
    dif[i]=abs(x[i]-y[i])

log_dif=np.log(dif)
a,b=np.polyfit(n,log_dif,1)
    
plt.figure()
plt.plot(n,log_dif,"C3.",markersize=4)
plt.plot(n,a*n+b)
plt.xlabel("N",fontsize=14)
plt.ylabel(r'$\Delta x$',fontsize=14)
plt.title("Trajectory  separation",fontsize=15)
plt.text(0,-2.5,"f(x)=a*x+b \na = "+str(a)+"\nb = "+str(b),fontsize=11)
plt.grid()
plt.savefig("2_1d_dif.pdf")
plt.show()
