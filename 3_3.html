def f(t,x,a=0.5):
    omega=1.5
    omega0=1.5
    gamma=3/8
    y=np.zeros(2)
    y[0]=x[1]
    y[1]=a*omega**2*np.sin(t)-omega0**2*np.sin(x[0])-2*gamma*x[1]
    return y

def rungeKutta4(t0,x0,tmax,h,a=0.5):

    N=(int)((tmax - t0)/h)
    x=np.zeros((N+1,2))
    t=np.zeros(N+1)
    x[0]=x0
    t[0]=t0
    for i in range(1,N+1):
    
        k1 = h * f(t[i-1],x[i-1],a)
        k2 = h * f(t[i-1]+0.5*h, x[i-1]+0.5*k1,a)
        k3 = h * f(t[i-1]+0.5*h, x[i-1]+0.5*k2,a)
        k4 = h * f(t[i-1]+h, x[i-1]+k3,a)
 
        x[i] = x[i-1] + (1.0 / 6.0)*(k1 + 2 * k2 + 2 * k3 + k4)
        t[i] = t[i-1] + h
    return x,t
